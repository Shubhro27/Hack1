def execute(pathToScript: String, jsonString: String, args: String*): Unit = {
    var counter = 1
    //Obtain Connection
    val jedis: Jedis = new Jedis(Properties.getRedisDBServer, Properties.getRedisDBPort.toInt, 2000);

    var script: String = null
    val source = scala.io.Source.fromFile(pathToScript)
    script = try source.mkString finally source.close()
    if (script != null && script.length > 0) {
      logger.info("executing script..")
      var keys = new ArrayList[String]();
      keys.add("key" + counter);
      counter = counter + 1

      var args = new ArrayList[String]();
      
args.add(jsonString)

      jedis.eval(script, keys, args)
      logger.info("Message Published Successfully")

    }
}
